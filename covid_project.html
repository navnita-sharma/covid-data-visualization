Top 10 Countries by Confirmed Cases

top10 <- country_data %>%
  arrange(desc(Confirmed)) %>%
  slice(1:10)

ggplot(top10, aes(x = reorder(`Country/Region`, Confirmed), 
                  y = Confirmed, fill = `Country/Region`)) +
  geom_bar(stat = "identity") +
  coord_flip() +
  labs(title = "Top 10 Countries by Confirmed Cases",
       x = "Country", y = "Confirmed Cases") +
  theme_minimal()

Stacked Bar Chart (Selected Countries)

india_us <- country_data %>%
  filter(`Country/Region` %in% c("India", "US","China","Canada","France")) %>%
  select(`Country/Region`, Confirmed, Deaths, Recovered) %>%
  pivot_longer(cols = c(Confirmed, Deaths, Recovered),
               names_to = "Category", values_to = "Count")

ggplot(india_us, aes(x = `Country/Region`, y = Count, fill = Category)) +
  geom_bar(stat = "identity", position = "stack") +
  labs(title = "COVID-19 Breakdown (Selected Countries)",
       x = "Country", y = "Cases") +
  theme_minimal()

Correlation Plot (Deaths vs Confirmed)

ggplot(country_data, aes(x = Confirmed, y = Deaths)) +
  geom_point(color = "darkblue", alpha = 0.5) +
  geom_smooth(method = "lm", se = FALSE, color = "red") +
  labs(title = "Correlation Between Confirmed Cases and Deaths",
       x = "Confirmed Cases", y = "Deaths") +
  theme_minimal()
